1. Otwieranie i zamykanie plików: ifstream, ofstream

W C++ do pracy z plikami używamy klas ifstream (do odczytu) i ofstream (do zapisu). Oto przykład otwierania i zamykania pliku:

#include <iostream>
#include <fstream>

using namespace std;

int main() {
    // Otwieranie pliku do zapisu
    ofstream plikDoZapisu("przykladowy.txt");
    if (plikDoZapisu.is_open()) {
        plikDoZapisu << "Witaj, świecie!" << endl;
        plikDoZapisu.close();  // Zamknięcie pliku
    } else {
        cout << "Nie można otworzyć pliku do zapisu." << endl;
    }

    // Otwieranie pliku do odczytu
    ifstream plikDoOdczytu("przykladowy.txt");
    if (plikDoOdczytu.is_open()) {
        string linia;
        while (getline(plikDoOdczytu, linia)) {
            cout << linia << endl;
        }
        plikDoOdczytu.close();  // Zamknięcie pliku
    } else {
        cout << "Nie można otworzyć pliku do odczytu." << endl;
    }

    return 0;
}

2. Tryby pracy z plikami: odczyt, zapis, dopisywanie

Pliki mogą być otwierane w różnych trybach: odczyt (in), zapis (out), oraz dopisywanie (app). Oto przykłady:

#include <iostream>
#include <fstream>

using namespace std;

int main() {
    // Tryb zapisu (nadpisanie pliku)
    ofstream plik("przykladowy.txt", out);
    plik << "Tekst do zapisu" << endl;
    plik.close();

    // Tryb dopisywania
    ofstream plikDopisywanie("przykladowy.txt", app);
    plikDopisywanie << "Tekst dopisany" << endl;
    plikDopisywanie.close();

    // Tryb odczytu
    ifstream plikOdczyt("przykladowy.txt", in);
    string linia;
    while (getline(plikOdczyt, linia)) {
        cout << linia << endl;
    }
    plikOdczyt.close();

    return 0;
}

3. Odczytywanie i zapisywanie danych: operacje podstawowe

Odczytywanie i zapisywanie danych w plikach można wykonać za pomocą podstawowych operacji, takich jak << dla zapisu i >> dla odczytu. Oto przykład:

#include <iostream>
#include <fstream>

using namespace std;

int main() {
    // Zapis danych do pliku
    ofstream plik("dane.txt");
    if (plik.is_open()) {
        int liczba = 42;
        plik << "Liczba: " << liczba << endl;
        plik.close();
    }

    // Odczyt danych z pliku
    ifstream plikOdczyt("dane.txt");
    if (plikOdczyt.is_open()) {
        string linia;
        while (getline(plikOdczyt, linia)) {
            cout << linia << endl;
        }
        plikOdczyt.close();
    }

    return 0;
}

4. Użycie RAII (Resource Acquisition Is Initialization) do zarządzania zasobami

RAII to technika zarządzania zasobami, w której zasoby są przydzielane podczas inicjalizacji obiektów i zwalniane podczas ich niszczenia. W przypadku plików, klasy ifstream i ofstream automatycznie zamykają pliki, gdy obiekty są niszczone:

#include <iostream>
#include <fstream>

using namespace std;

void funkcja() {
    ofstream plik("przykladowy.txt");
    plik << "Przykład RAII" << endl;
    // Plik zostanie automatycznie zamknięty, gdy obiekt plik zostanie zniszczony
}

int main() {
    funkcja();

    ifstream plik("przykladowy.txt");
    string linia;
    while (getline(plik, linia)) {
        cout << linia << endl;
    }

    return 0;
}
