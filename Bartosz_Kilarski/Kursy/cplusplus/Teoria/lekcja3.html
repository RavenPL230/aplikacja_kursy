    1. Instrukcje warunkowe: `if`, `else if`, `else`:

    Instrukcje warunkowe służą do podejmowania decyzji na podstawie spełnionych warunków. Program wykonuje określony blok kodu, jeśli warunek jest prawdziwy.

    #include <iostream>

using namespace std;

int main() {
    int x = 10;

    if (x > 0) {
        cout << "x jest dodatnia" << endl;  // Wykonane, gdy x jest większe od 0
    } else if (x < 0) {
        cout << "x jest ujemna" << endl;  // Wykonane, gdy x jest mniejsze od 0
    } else {
        cout << "x jest równa zero" << endl;  // Wykonane, gdy x jest równe 0
    }

    return 0;
}

    2. Pętle: `for`, `while`, `do-while`:

    Pętle pozwalają na wielokrotne wykonywanie tego samego bloku kodu. Typ pętli zależy od warunków jej zakończenia.

    #include <iostream>

using namespace std;

int main() {
    // Pętla for
    for (int i = 0; i < 5; ++i) {
        cout << i << " ";
    }
    cout << endl;

    // Pętla while
    int j = 0;
    while (j < 5) {
        cout << j << " ";
        ++j;
    }
    cout << endl;

    // Pętla do-while
    int k = 0;
    do {
        cout << k << " ";
        ++k;
    } while (k < 5);
    cout << endl;

    return 0;
}

    3. Instrukcje sterujące: `break`, `continue`:

    Instrukcje sterujące `break` i `continue` służą do zmiany przebiegu pętli. `break` przerywa pętlę, podczas gdy `continue` pomija resztę kodu w danej iteracji i przechodzi do następnej.

    #include <iostream>

using namespace std;

int main() {
    // Przykład użycia break
    for (int i = 0; i < 10; ++i) {
        if (i == 5) {
            break;  // Przerywa pętlę, gdy i jest równe 5
        }
        cout << i << " ";
    }
    cout << endl;

    // Przykład użycia continue
    for (int i = 0; i < 10; ++i) {
        if (i % 2 == 0) {
            continue;  // Pomija resztę kodu w pętli, gdy i jest liczbą parzystą
        }
        cout << i << " ";
    }
    cout << endl;

    return 0;
}

    4. Iteratory: podstawowe operacje i iteracja po kolekcjach:

    Iteratory umożliwiają przechodzenie przez elementy kolekcji, takich jak wektory i listy. Oto przykłady podstawowych operacji:

    #include <iostream>
#include <vector>
#include <list>

using namespace std;

int main() {
    // Iteracja po wektorze
    vector vec = {1, 2, 3, 4, 5};
    for (vector::iterator it = vec.begin(); it != vec.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    // Iteracja po liście
    list lst = {1, 2, 3, 4, 5};
    for (list::iterator it = lst.begin(); it != lst.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    return 0;
}
